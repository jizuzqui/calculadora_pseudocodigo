/*
* Función que suma dos números enteros
* Params a, b
*/
int suma (int a, int b) {

        return a + b;
}

/*
* Función que resta dos números enteros
* Params a, b
*/
int resta (int a, int b) {

        return a - b;
}

/*
* Función que multiplica dos números enteros
* Params a, b
*/
int multiplicacion (int a, int b) {

        return a * b;
}
